local Library = loadstring(game:HttpGet("https://raw.githubusercontent.com/xHeptc/Kavo-UI-Library/main/source.lua"))()

local Window = Library.CreateLib("Whale Hub | Slap Battles [ADMIN]", "Serpent")

local Tab = Window:NewTab("[ ! COMMANDS ! ] ")

local Section = Tab:NewSection("This Script Is A Tribute To Aquoupi")

local Section = Tab:NewSection("Free Abilitys / [1]")

Section:NewButton(":Null", "Uses Nulls Ability", function()
    game:GetService("ReplicatedStorage").NullAbility:FireServer()
end)

Section:NewButton(":Rhythm", "Uses Rhythms Explosion", function()
    game:GetService("ReplicatedStorage").rhythmevent:FireServer("AoeExplosion",0)
    
end)

local Section = Tab:NewSection("Slap Require Commands / [2] ")

Section:NewButton(":Diamond", "Uses Ability", function()
    fireclickdetector(workspace.Lobby.Diamond.ClickDetector)
    wait(0.01)
game:GetService("ReplicatedStorage"):WaitForChild("Rockmode"):FireServer()

end)

Section:NewButton(":Sleep", "Uses Ability", function()
    fireclickdetector(workspace.Lobby.ZZZZZZZ.ClickDetector)
    wait(0.01)
game:GetService("ReplicatedStorage"):WaitForChild("ZZZZZZZSleep"):FireServer()

end)
